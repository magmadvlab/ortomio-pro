-- Funzione per aggiornare updated_at automaticamente
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ language 'plpgsql';

-- Trigger per aggiornare updated_at su cultivation_plans
DROP TRIGGER IF EXISTS update_cultivation_plans_updated_at ON cultivation_plans;
CREATE TRIGGER update_cultivation_plans_updated_at
    BEFORE UPDATE ON cultivation_plans
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

-- Trigger per aggiornare updated_at su cultivation_issues
DROP TRIGGER IF EXISTS update_cultivation_issues_updated_at ON cultivation_issues;
CREATE TRIGGER update_cultivation_issues_updated_at
    BEFORE UPDATE ON cultivation_issues
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

-- Trigger per aggiornare updated_at su sapling_inventory
DROP TRIGGER IF EXISTS update_sapling_inventory_updated_at ON sapling_inventory;
CREATE TRIGGER update_sapling_inventory_updated_at
    BEFORE UPDATE ON sapling_inventory
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

-- Funzione semplificata per conteggio piante
CREATE OR REPLACE FUNCTION update_plant_count_simple()
RETURNS TRIGGER AS $$
BEGIN
    -- Aggiorna il conteggio delle piante nel piano di coltivazione
    IF TG_OP = 'DELETE' THEN
        UPDATE cultivation_plans 
        SET actual_plants_count = (
            SELECT COUNT(*) 
            FROM sapling_inventory 
            WHERE cultivation_plan_id = OLD.cultivation_plan_id
            AND health_status IN ('healthy', 'stressed')
        )
        WHERE id = OLD.cultivation_plan_id;
        RETURN OLD;
    ELSE
        UPDATE cultivation_plans 
        SET actual_plants_count = (
            SELECT COUNT(*) 
            FROM sapling_inventory 
            WHERE cultivation_plan_id = NEW.cultivation_plan_id
            AND health_status IN ('healthy', 'stressed')
        )
        WHERE id = NEW.cultivation_plan_id;
        RETURN NEW;
    END IF;
END;
$$ LANGUAGE plpgsql;

-- Trigger per aggiornamento conteggio piante
DROP TRIGGER IF EXISTS update_cultivation_plant_count ON sapling_inventory;
CREATE TRIGGER update_cultivation_plant_count
    AFTER INSERT OR UPDATE OR DELETE ON sapling_inventory
    FOR EACH ROW
    EXECUTE FUNCTION update_plant_count_simple();

-- Funzione per validazione date semplificata
CREATE OR REPLACE FUNCTION validate_cultivation_dates_simple()
RETURNS TRIGGER AS $$
BEGIN
    -- Valida che le date siano logiche
    IF NEW.actual_end_date IS NOT NULL AND NEW.actual_start_date IS NOT NULL THEN
        IF NEW.actual_end_date < NEW.actual_start_date THEN
            RAISE EXCEPTION 'End date cannot be before start date';
        END IF;
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger per validazione date
DROP TRIGGER IF EXISTS validate_cultivation_plan_dates ON cultivation_plans;
CREATE TRIGGER validate_cultivation_plan_dates
    BEFORE INSERT OR UPDATE ON cultivation_plans
    FOR EACH ROW
    EXECUTE FUNCTION validate_cultivation_dates_simple();

-- Funzione per logging semplice delle transizioni
CREATE OR REPLACE FUNCTION log_status_change()
RETURNS TRIGGER AS $$
BEGIN
    -- Log cambio di status
    IF OLD.status IS DISTINCT FROM NEW.status THEN
        INSERT INTO phase_transitions (
            cultivation_plan_id,
            from_phase,
            to_phase,
            transition_type,
            trigger_condition,
            actual_date
        ) VALUES (
            NEW.id,
            COALESCE(OLD.status, 'new'),
            NEW.status,
            'manual',
            'status_change',
            CURRENT_DATE
        );
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger per logging cambio status
DROP TRIGGER IF EXISTS log_cultivation_status_change ON cultivation_plans;
CREATE TRIGGER log_cultivation_status_change
    AFTER UPDATE ON cultivation_plans
    FOR EACH ROW
    EXECUTE FUNCTION log_status_change();

-- Funzione per calcolo statistiche base
CREATE OR REPLACE FUNCTION update_basic_statistics()
RETURNS TRIGGER AS $$
DECLARE
    total_harvest DECIMAL(10,2);
    harvest_count INTEGER;
BEGIN
    -- Calcola totale raccolti per il piano
    SELECT 
        COALESCE(SUM(quantity_kg), 0),
        COUNT(*)
    INTO total_harvest, harvest_count
    FROM detailed_harvests 
    WHERE cultivation_plan_id = NEW.cultivation_plan_id;
    
    -- Aggiorna o inserisci statistiche base
    INSERT INTO cultivation_statistics (
        cultivation_plan_id,
        actual_yield_kg,
        measurement_date
    ) VALUES (
        NEW.cultivation_plan_id,
        total_harvest,
        CURRENT_DATE
    )
    ON CONFLICT (cultivation_plan_id, measurement_date) 
    DO UPDATE SET
        actual_yield_kg = EXCLUDED.actual_yield_kg;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger per aggiornamento statistiche base
DROP TRIGGER IF EXISTS update_harvest_statistics ON detailed_harvests;
CREATE TRIGGER update_harvest_statistics
    AFTER INSERT OR UPDATE ON detailed_harvests
    FOR EACH ROW
    EXECUTE FUNCTION update_basic_statistics();
