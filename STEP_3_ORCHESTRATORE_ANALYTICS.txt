-- Tabella per dashboard analytics
CREATE TABLE IF NOT EXISTS cultivation_analytics_dashboard (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
    
    -- Periodo di analisi
    analysis_period VARCHAR(20) CHECK (analysis_period IN ('daily', 'weekly', 'monthly', 'seasonal', 'yearly')),
    period_start_date DATE NOT NULL,
    period_end_date DATE NOT NULL,
    
    -- Metriche generali
    total_active_plans INTEGER DEFAULT 0,
    total_completed_plans INTEGER DEFAULT 0,
    total_area_m2 DECIMAL(10,2) DEFAULT 0,
    total_plants_count INTEGER DEFAULT 0,
    
    -- Metriche produzione
    total_harvest_kg DECIMAL(10,2) DEFAULT 0,
    average_yield_per_m2 DECIMAL(8,2) DEFAULT 0,
    production_efficiency DECIMAL(5,2) DEFAULT 0, -- percentuale rispetto al target
    
    -- Metriche qualità
    average_quality_score DECIMAL(5,2) DEFAULT 0,
    premium_grade_percentage DECIMAL(5,2) DEFAULT 0,
    
    -- Metriche problemi
    total_issues_count INTEGER DEFAULT 0,
    critical_issues_count INTEGER DEFAULT 0,
    resolution_rate DECIMAL(5,2) DEFAULT 0, -- percentuale problemi risolti
    
    -- Metriche risorse
    total_water_consumption DECIMAL(10,2) DEFAULT 0,
    total_fertilizer_consumption DECIMAL(8,2) DEFAULT 0,
    total_labor_hours DECIMAL(8,2) DEFAULT 0,
    
    -- Metriche economiche
    estimated_production_value DECIMAL(10,2) DEFAULT 0,
    cost_per_kg DECIMAL(8,2) DEFAULT 0,
    roi_percentage DECIMAL(5,2) DEFAULT 0,
    
    -- Metadati
    calculated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    UNIQUE(user_id, analysis_period, period_start_date)
);

-- Tabella per dashboard principale
CREATE TABLE IF NOT EXISTS cultivation_dashboard (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
    
    -- Configurazione dashboard
    dashboard_name VARCHAR(255) NOT NULL DEFAULT 'My Garden Dashboard',
    layout_config JSONB DEFAULT '{}', -- configurazione layout widgets
    
    -- Widget attivi
    active_widgets JSONB DEFAULT '[]', -- lista widget abilitati
    widget_positions JSONB DEFAULT '{}', -- posizioni widget nel layout
    
    -- Filtri e preferenze
    default_time_range VARCHAR(50) DEFAULT 'last_30_days',
    favorite_metrics JSONB DEFAULT '[]', -- metriche preferite dell'utente
    alert_thresholds JSONB DEFAULT '{}', -- soglie per alert personalizzati
    
    -- Personalizzazione
    color_scheme VARCHAR(50) DEFAULT 'default',
    chart_preferences JSONB DEFAULT '{}', -- preferenze grafici
    
    -- Metadati
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    last_accessed_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    UNIQUE(user_id, dashboard_name)
);

-- Funzione per calcolare analytics dashboard
CREATE OR REPLACE FUNCTION calculate_cultivation_analytics(
    p_user_id UUID,
    p_period VARCHAR(20),
    p_start_date DATE,
    p_end_date DATE
)
RETURNS void AS $$
DECLARE
    analytics_record cultivation_analytics_dashboard%ROWTYPE;
    total_plans INTEGER;
    completed_plans INTEGER;
    active_plans INTEGER;
    total_area DECIMAL(10,2);
    total_plants INTEGER;
    total_harvest DECIMAL(10,2);
    avg_yield DECIMAL(8,2);
    avg_quality DECIMAL(5,2);
    premium_percentage DECIMAL(5,2);
    total_issues INTEGER;
    critical_issues INTEGER;
    resolved_issues INTEGER;
    resolution_rate DECIMAL(5,2);
    total_water DECIMAL(10,2);
    total_fertilizer DECIMAL(8,2);
    total_labor DECIMAL(8,2);
    estimated_value DECIMAL(10,2);
BEGIN
    -- Calcola metriche base
    SELECT 
        COUNT(*),
        COUNT(*) FILTER (WHERE status = 'completed'),
        COUNT(*) FILTER (WHERE status = 'active'),
        COALESCE(SUM(area_m2), 0),
        COALESCE(SUM(actual_plants_count), 0)
    INTO total_plans, completed_plans, active_plans, total_area, total_plants
    FROM cultivation_plans 
    WHERE user_id = p_user_id
    AND created_at::date BETWEEN p_start_date AND p_end_date;
    
    -- Calcola metriche raccolto
    SELECT 
        COALESCE(SUM(dh.quantity_kg), 0),
        COALESCE(AVG(CASE dh.quality_grade 
            WHEN 'premium' THEN 10
            WHEN 'standard' THEN 8
            WHEN 'seconds' THEN 6
            ELSE 4
        END), 0),
        COALESCE(AVG(dh.estimated_value), 0)
    INTO total_harvest, avg_quality, estimated_value
    FROM detailed_harvests dh
    JOIN cultivation_plans cp ON cp.id = dh.cultivation_plan_id
    WHERE cp.user_id = p_user_id
    AND dh.harvest_date BETWEEN p_start_date AND p_end_date;
    
    -- Calcola yield per m2
    IF total_area > 0 THEN
        avg_yield := total_harvest / total_area;
    ELSE
        avg_yield := 0;
    END IF;
    
    -- Calcola percentuale premium
    SELECT 
        COALESCE(
            (COUNT(*) FILTER (WHERE quality_grade = 'premium')::DECIMAL / NULLIF(COUNT(*), 0)) * 100,
            0
        )
    INTO premium_percentage
    FROM detailed_harvests dh
    JOIN cultivation_plans cp ON cp.id = dh.cultivation_plan_id
    WHERE cp.user_id = p_user_id
    AND dh.harvest_date BETWEEN p_start_date AND p_end_date;
    
    -- Calcola metriche problemi
    SELECT 
        COUNT(*),
        COUNT(*) FILTER (WHERE severity = 'critical'),
        COUNT(*) FILTER (WHERE status = 'resolved')
    INTO total_issues, critical_issues, resolved_issues
    FROM cultivation_issues ci
    JOIN cultivation_plans cp ON cp.id = ci.cultivation_plan_id
    WHERE cp.user_id = p_user_id
    AND ci.detected_date BETWEEN p_start_date AND p_end_date;
    
    -- Calcola tasso risoluzione
    IF total_issues > 0 THEN
        resolution_rate := (resolved_issues::DECIMAL / total_issues) * 100;
    ELSE
        resolution_rate := 100;
    END IF;
    
    -- Calcola metriche risorse
    SELECT 
        COALESCE(SUM(water_consumption_liters), 0),
        COALESCE(SUM(fertilizer_consumption_kg), 0),
        COALESCE(SUM(labor_hours), 0)
    INTO total_water, total_fertilizer, total_labor
    FROM cultivation_statistics cs
    JOIN cultivation_plans cp ON cp.id = cs.cultivation_plan_id
    WHERE cp.user_id = p_user_id
    AND cs.measurement_date BETWEEN p_start_date AND p_end_date;
    
    -- Inserisci o aggiorna analytics
    INSERT INTO cultivation_analytics_dashboard (
        user_id,
        analysis_period,
        period_start_date,
        period_end_date,
        total_active_plans,
        total_completed_plans,
        total_area_m2,
        total_plants_count,
        total_harvest_kg,
        average_yield_per_m2,
        average_quality_score,
        premium_grade_percentage,
        total_issues_count,
        critical_issues_count,
        resolution_rate,
        total_water_consumption,
        total_fertilizer_consumption,
        total_labor_hours,
        estimated_production_value
    ) VALUES (
        p_user_id,
        p_period,
        p_start_date,
        p_end_date,
        active_plans,
        completed_plans,
        total_area,
        total_plants,
        total_harvest,
        avg_yield,
        avg_quality,
        premium_percentage,
        total_issues,
        critical_issues,
        resolution_rate,
        total_water,
        total_fertilizer,
        total_labor,
        estimated_value
    )
    ON CONFLICT (user_id, analysis_period, period_start_date)
    DO UPDATE SET
        total_active_plans = EXCLUDED.total_active_plans,
        total_completed_plans = EXCLUDED.total_completed_plans,
        total_area_m2 = EXCLUDED.total_area_m2,
        total_plants_count = EXCLUDED.total_plants_count,
        total_harvest_kg = EXCLUDED.total_harvest_kg,
        average_yield_per_m2 = EXCLUDED.average_yield_per_m2,
        average_quality_score = EXCLUDED.average_quality_score,
        premium_grade_percentage = EXCLUDED.premium_grade_percentage,
        total_issues_count = EXCLUDED.total_issues_count,
        critical_issues_count = EXCLUDED.critical_issues_count,
        resolution_rate = EXCLUDED.resolution_rate,
        total_water_consumption = EXCLUDED.total_water_consumption,
        total_fertilizer_consumption = EXCLUDED.total_fertilizer_consumption,
        total_labor_hours = EXCLUDED.total_labor_hours,
        estimated_production_value = EXCLUDED.estimated_production_value,
        calculated_at = NOW();
        
END;
$$ LANGUAGE plpgsql;

-- Funzione per ottenere metriche dashboard in tempo reale
CREATE OR REPLACE FUNCTION get_dashboard_metrics(p_user_id UUID)
RETURNS TABLE (
    metric_name VARCHAR(100),
    metric_value DECIMAL(10,2),
    metric_unit VARCHAR(20),
    trend_direction VARCHAR(10),
    trend_percentage DECIMAL(5,2)
) AS $$
BEGIN
    RETURN QUERY
    WITH current_metrics AS (
        SELECT 
            'active_plans' as name,
            COUNT(*)::DECIMAL as value,
            'count' as unit
        FROM cultivation_plans 
        WHERE user_id = p_user_id AND status = 'active'
        
        UNION ALL
        
        SELECT 
            'total_harvest_this_month',
            COALESCE(SUM(dh.quantity_kg), 0),
            'kg'
        FROM detailed_harvests dh
        JOIN cultivation_plans cp ON cp.id = dh.cultivation_plan_id
        WHERE cp.user_id = p_user_id
        AND dh.harvest_date >= DATE_TRUNC('month', CURRENT_DATE)
        
        UNION ALL
        
        SELECT 
            'average_quality_score',
            COALESCE(AVG(CASE dh.quality_grade 
                WHEN 'premium' THEN 10
                WHEN 'standard' THEN 8
                WHEN 'seconds' THEN 6
                ELSE 4
            END), 0),
            'score'
        FROM detailed_harvests dh
        JOIN cultivation_plans cp ON cp.id = dh.cultivation_plan_id
        WHERE cp.user_id = p_user_id
        AND dh.harvest_date >= CURRENT_DATE - INTERVAL '30 days'
        
        UNION ALL
        
        SELECT 
            'open_issues',
            COUNT(*)::DECIMAL,
            'count'
        FROM cultivation_issues ci
        JOIN cultivation_plans cp ON cp.id = ci.cultivation_plan_id
        WHERE cp.user_id = p_user_id
        AND ci.status IN ('open', 'in_progress')
    )
    SELECT 
        cm.name::VARCHAR(100),
        cm.value,
        cm.unit::VARCHAR(20),
        'stable'::VARCHAR(10), -- Placeholder per trend
        0::DECIMAL(5,2) -- Placeholder per percentuale trend
    FROM current_metrics cm;
END;
$$ LANGUAGE plpgsql;

-- Indici per performance analytics
CREATE INDEX IF NOT EXISTS idx_cultivation_analytics_user_period ON cultivation_analytics_dashboard(user_id, analysis_period, period_start_date);
CREATE INDEX IF NOT EXISTS idx_cultivation_dashboard_user ON cultivation_dashboard(user_id);
CREATE INDEX IF NOT EXISTS idx_cultivation_dashboard_accessed ON cultivation_dashboard(last_accessed_at);

-- RLS Policies per analytics
ALTER TABLE cultivation_analytics_dashboard ENABLE ROW LEVEL SECURITY;
ALTER TABLE cultivation_dashboard ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can manage their own analytics" ON cultivation_analytics_dashboard
    FOR ALL USING (auth.uid() = user_id);

CREATE POLICY "Users can manage their own dashboard" ON cultivation_dashboard
    FOR ALL USING (auth.uid() = user_id);

-- Trigger per aggiornare dashboard access time
CREATE OR REPLACE FUNCTION update_dashboard_access()
RETURNS TRIGGER AS $$
BEGIN
    NEW.last_accessed_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS update_dashboard_access_time ON cultivation_dashboard;
CREATE TRIGGER update_dashboard_access_time
    BEFORE UPDATE ON cultivation_dashboard
    FOR EACH ROW
    EXECUTE FUNCTION update_dashboard_access();

-- Funzione per generare report settimanale automatico
CREATE OR REPLACE FUNCTION generate_weekly_analytics_report(p_user_id UUID)
RETURNS void AS $$
DECLARE
    week_start DATE;
    week_end DATE;
BEGIN
    -- Calcola inizio e fine settimana corrente
    week_start := DATE_TRUNC('week', CURRENT_DATE);
    week_end := week_start + INTERVAL '6 days';
    
    -- Genera analytics per la settimana
    PERFORM calculate_cultivation_analytics(
        p_user_id,
        'weekly',
        week_start,
        week_end
    );
END;
$$ LANGUAGE plpgsql;
