-- ============================================
-- ORTOMIO - ORCHESTRATORE COLTIVAZIONE (SENZA FK)
-- Aggiunge tabelle per gestire il ciclo completo
-- Senza foreign key per evitare conflitti di tipo
-- ============================================

-- Tabella principale per i piani di coltivazione
CREATE TABLE IF NOT EXISTS cultivation_plans (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
    garden_id UUID REFERENCES gardens(id) ON DELETE CASCADE,
    
    -- Riferimenti senza FK (per evitare conflitti di tipo)
    crop_archetype_id TEXT, -- Riferimento a crop_archetypes.id
    crop_profile_id TEXT,   -- Riferimento a crop_profiles.id (se esiste)
    
    -- Informazioni base
    name VARCHAR(255) NOT NULL,
    description TEXT,
    status VARCHAR(50) DEFAULT 'planned' CHECK (status IN ('planned', 'active', 'completed', 'cancelled')),
    
    -- Date pianificazione
    planned_start_date DATE,
    actual_start_date DATE,
    planned_end_date DATE,
    actual_end_date DATE,
    
    -- Configurazione spazio
    area_m2 DECIMAL(10,2),
    expected_plants_count INTEGER,
    actual_plants_count INTEGER DEFAULT 0,
    
    -- Configurazione irrigazione
    irrigation_method VARCHAR(100),
    irrigation_frequency_days INTEGER DEFAULT 3,
    water_amount_liters_per_m2 DECIMAL(8,2),
    
    -- Configurazione nutrizione
    fertilization_schedule JSONB DEFAULT '[]',
    soil_preparation_notes TEXT,
    
    -- Metadati
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    -- Constraint per date
    CONSTRAINT cultivation_plans_dates_check CHECK (
        planned_start_date IS NULL OR planned_end_date IS NULL OR planned_start_date <= planned_end_date
    )
);

-- Tabella per statistiche e metriche
CREATE TABLE IF NOT EXISTS cultivation_statistics (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    cultivation_plan_id UUID NOT NULL REFERENCES cultivation_plans(id) ON DELETE CASCADE,
    
    -- Metriche crescita
    germination_rate DECIMAL(5,2), -- Percentuale germinazione
    survival_rate DECIMAL(5,2), -- Percentuale sopravvivenza
    growth_rate_cm_per_week DECIMAL(8,2),
    
    -- Metriche produzione
    expected_yield_kg DECIMAL(10,2),
    actual_yield_kg DECIMAL(10,2),
    yield_efficiency DECIMAL(5,2), -- actual/expected * 100
    
    -- Metriche risorse
    water_consumption_liters DECIMAL(10,2),
    fertilizer_consumption_kg DECIMAL(8,2),
    labor_hours DECIMAL(8,2),
    
    -- Metriche qualità
    quality_score INTEGER CHECK (quality_score BETWEEN 1 AND 10),
    pest_incidents_count INTEGER DEFAULT 0,
    disease_incidents_count INTEGER DEFAULT 0,
    
    -- Metadati
    measurement_date DATE NOT NULL,
    notes TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    UNIQUE(cultivation_plan_id, measurement_date)
);

-- Tabella per tracking problemi e soluzioni
CREATE TABLE IF NOT EXISTS cultivation_issues (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    cultivation_plan_id UUID NOT NULL REFERENCES cultivation_plans(id) ON DELETE CASCADE,
    
    -- Classificazione problema
    issue_type VARCHAR(100) NOT NULL CHECK (issue_type IN (
        'pest', 'disease', 'nutrient_deficiency', 'water_stress', 
        'weather_damage', 'soil_problem', 'growth_anomaly', 'other'
    )),
    severity VARCHAR(20) DEFAULT 'medium' CHECK (severity IN ('low', 'medium', 'high', 'critical')),
    
    -- Descrizione
    title VARCHAR(255) NOT NULL,
    description TEXT,
    symptoms TEXT,
    
    -- Localizzazione
    affected_area_percentage DECIMAL(5,2),
    location_notes TEXT,
    
    -- Gestione
    status VARCHAR(50) DEFAULT 'open' CHECK (status IN ('open', 'in_progress', 'resolved', 'monitoring')),
    solution_applied TEXT,
    treatment_date DATE,
    resolution_date DATE,
    
    -- Prevenzione futura
    prevention_notes TEXT,
    recurrence_risk VARCHAR(20) CHECK (recurrence_risk IN ('low', 'medium', 'high')),
    
    -- Metadati
    detected_date DATE NOT NULL DEFAULT CURRENT_DATE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Tabella per raccolti dettagliati
CREATE TABLE IF NOT EXISTS detailed_harvests (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    cultivation_plan_id UUID NOT NULL REFERENCES cultivation_plans(id) ON DELETE CASCADE,
    
    -- Informazioni raccolto
    harvest_date DATE NOT NULL,
    quantity_kg DECIMAL(10,3) NOT NULL,
    quality_grade VARCHAR(20) CHECK (quality_grade IN ('premium', 'standard', 'seconds', 'compost')),
    
    -- Classificazione prodotto
    size_category VARCHAR(50), -- small, medium, large, extra_large
    ripeness_level VARCHAR(50), -- unripe, perfect, overripe
    
    -- Condizioni raccolto
    weather_conditions VARCHAR(100),
    harvester_notes TEXT,
    
    -- Destinazione
    destination VARCHAR(100) CHECK (destination IN ('consumption', 'storage', 'sale', 'gift', 'compost', 'seeds')),
    storage_method VARCHAR(100),
    expected_storage_days INTEGER,
    
    -- Valutazione economica
    market_price_per_kg DECIMAL(8,2),
    estimated_value DECIMAL(10,2),
    
    -- Metadati
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Tabella per transizioni automatiche delle fasi
CREATE TABLE IF NOT EXISTS phase_transitions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    cultivation_plan_id UUID NOT NULL REFERENCES cultivation_plans(id) ON DELETE CASCADE,
    
    -- Fase di transizione
    from_phase VARCHAR(100),
    to_phase VARCHAR(100) NOT NULL,
    transition_type VARCHAR(50) CHECK (transition_type IN ('automatic', 'manual', 'scheduled')),
    
    -- Trigger della transizione
    trigger_condition VARCHAR(200), -- es: "days_since_planting >= 14"
    trigger_value JSONB, -- valore che ha scatenato la transizione
    
    -- Timing
    scheduled_date DATE,
    actual_date DATE NOT NULL DEFAULT CURRENT_DATE,
    
    -- Azioni automatiche eseguite
    actions_performed JSONB DEFAULT '[]', -- lista azioni automatiche
    notifications_sent JSONB DEFAULT '[]', -- notifiche inviate
    
    -- Metadati
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    notes TEXT
);

-- Tabella per inventario piantine
CREATE TABLE IF NOT EXISTS sapling_inventory (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    cultivation_plan_id UUID REFERENCES cultivation_plans(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
    
    -- Identificazione piantina (senza FK)
    crop_archetype_id TEXT, -- Riferimento a crop_archetypes.id
    variety_name VARCHAR(200),
    source VARCHAR(100) CHECK (source IN ('seed', 'cutting', 'transplant', 'purchase')),
    
    -- Stato fisico
    current_phase VARCHAR(100) DEFAULT 'seedling',
    health_status VARCHAR(50) DEFAULT 'healthy' CHECK (health_status IN ('healthy', 'stressed', 'sick', 'dying', 'dead')),
    size_cm DECIMAL(8,2),
    
    -- Localizzazione
    location_type VARCHAR(50) CHECK (location_type IN ('greenhouse', 'indoor', 'outdoor', 'nursery')),
    container_type VARCHAR(100),
    position_notes TEXT,
    
    -- Date importanti
    germination_date DATE,
    transplant_ready_date DATE,
    actual_transplant_date DATE,
    
    -- Cure ricevute
    last_watering_date DATE,
    last_fertilization_date DATE,
    treatments_applied JSONB DEFAULT '[]',
    
    -- Metadati
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Indici per performance
CREATE INDEX IF NOT EXISTS idx_cultivation_plans_user_id ON cultivation_plans(user_id);
CREATE INDEX IF NOT EXISTS idx_cultivation_plans_status ON cultivation_plans(status);
CREATE INDEX IF NOT EXISTS idx_cultivation_plans_dates ON cultivation_plans(planned_start_date, planned_end_date);
CREATE INDEX IF NOT EXISTS idx_cultivation_plans_archetype ON cultivation_plans(crop_archetype_id);
CREATE INDEX IF NOT EXISTS idx_cultivation_statistics_plan_id ON cultivation_statistics(cultivation_plan_id);
CREATE INDEX IF NOT EXISTS idx_cultivation_issues_plan_id ON cultivation_issues(cultivation_plan_id);
CREATE INDEX IF NOT EXISTS idx_cultivation_issues_status ON cultivation_issues(status);
CREATE INDEX IF NOT EXISTS idx_detailed_harvests_plan_id ON detailed_harvests(cultivation_plan_id);
CREATE INDEX IF NOT EXISTS idx_detailed_harvests_date ON detailed_harvests(harvest_date);
CREATE INDEX IF NOT EXISTS idx_phase_transitions_plan_id ON phase_transitions(cultivation_plan_id);
CREATE INDEX IF NOT EXISTS idx_sapling_inventory_user_id ON sapling_inventory(user_id);
CREATE INDEX IF NOT EXISTS idx_sapling_inventory_plan_id ON sapling_inventory(cultivation_plan_id);

-- RLS Policies
ALTER TABLE cultivation_plans ENABLE ROW LEVEL SECURITY;
ALTER TABLE cultivation_statistics ENABLE ROW LEVEL SECURITY;
ALTER TABLE cultivation_issues ENABLE ROW LEVEL SECURITY;
ALTER TABLE detailed_harvests ENABLE ROW LEVEL SECURITY;
ALTER TABLE phase_transitions ENABLE ROW LEVEL SECURITY;
ALTER TABLE sapling_inventory ENABLE ROW LEVEL SECURITY;

-- Policy per cultivation_plans
CREATE POLICY "Users can manage their own cultivation plans" ON cultivation_plans
    FOR ALL USING (auth.uid() = user_id);

-- Policy per cultivation_statistics
CREATE POLICY "Users can manage statistics of their cultivation plans" ON cultivation_statistics
    FOR ALL USING (
        EXISTS (
            SELECT 1 FROM cultivation_plans 
            WHERE cultivation_plans.id = cultivation_statistics.cultivation_plan_id 
            AND cultivation_plans.user_id = auth.uid()
        )
    );

-- Policy per cultivation_issues
CREATE POLICY "Users can manage issues of their cultivation plans" ON cultivation_issues
    FOR ALL USING (
        EXISTS (
            SELECT 1 FROM cultivation_plans 
            WHERE cultivation_plans.id = cultivation_issues.cultivation_plan_id 
            AND cultivation_plans.user_id = auth.uid()
        )
    );

-- Policy per detailed_harvests
CREATE POLICY "Users can manage harvests of their cultivation plans" ON detailed_harvests
    FOR ALL USING (
        EXISTS (
            SELECT 1 FROM cultivation_plans 
            WHERE cultivation_plans.id = detailed_harvests.cultivation_plan_id 
            AND cultivation_plans.user_id = auth.uid()
        )
    );

-- Policy per phase_transitions
CREATE POLICY "Users can view transitions of their cultivation plans" ON phase_transitions
    FOR ALL USING (
        EXISTS (
            SELECT 1 FROM cultivation_plans 
            WHERE cultivation_plans.id = phase_transitions.cultivation_plan_id 
            AND cultivation_plans.user_id = auth.uid()
        )
    );

-- Policy per sapling_inventory
CREATE POLICY "Users can manage their own sapling inventory" ON sapling_inventory
    FOR ALL USING (auth.uid() = user_id);

COMMENT ON TABLE cultivation_plans IS 'Piani di coltivazione con automazione completa';

COMMENT ON TABLE cultivation_issues IS 'Tracking problemi e soluzioni per le coltivazioni';
COMMENT ON TABLE detailed_harvests IS 'Raccolti dettagliati con classificazione qualità';
COMMENT ON TABLE phase_transitions IS 'Transizioni automatiche delle fasi di crescita';
COMMENT ON TABLE sapling_inventory IS 'Inventario piantine con tracking completo';
